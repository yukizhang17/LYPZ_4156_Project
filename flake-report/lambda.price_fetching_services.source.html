74 # reject outlier data points which are likely not the target product
75 def reject_outliers(data, m=2.):
76 # choose the median of the first 10 items as reference sample
77 d = np.abs(data - np.median(data[:10]))
78 mdev = np.median(d)
79 s = d / (mdev if mdev else 1.)
80 res = data[s < m]
81 if len(res) > 0 and isinstance(res[0], np.ndarray):
82 res = res[0]
83 return res
